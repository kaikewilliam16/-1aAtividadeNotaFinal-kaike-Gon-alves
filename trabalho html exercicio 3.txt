function validarFormulario() {
    var nome = document.getElementById("nome").value;
    var email = document.getElementById("email").value;
    var mensagem = document.getElementById("mensagem").value;

    var statusMessage = document.getElementById("mensagem-status");

    statusMessage.innerHTML = "";
    statusMessage.classList.remove("sucesso", "erro");

    if (nome === "" || email === "" || mensagem === "") {
        statusMessage.innerHTML = "Todos os campos são obrigatórios!";
        statusMessage.classList.add("erro");
        return false;
    }

    var emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(email)) {
        statusMessage.innerHTML = "Por favor, insira um email válido.";
        statusMessage.classList.add("erro");
        return false;
    }

    statusMessage.innerHTML = "Formulário enviado com sucesso!";
    statusMessage.classList.add("sucesso");

    document.getElementById("form-contato").reset();
