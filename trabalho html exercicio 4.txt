javascript

function calcularIMC() {
   
    var peso = parseFloat(document.getElementById("peso").value);
    var altura = parseFloat(document.getElementById("altura").value);

   
    var resultadoDiv = document.getElementById("resultado");

    
    if (isNaN(peso) || isNaN(altura) || peso <= 0 || altura <= 0) {
        resultadoDiv.innerHTML = "Por favor, insira valores válidos para peso e altura.";
        resultadoDiv.classList.add("imc-alerta");
        resultadoDiv.classList.remove("imc-ok", "imc-perigo");
        return;
    }

    var imc = peso / (altura * altura);

    resultadoDiv.classList.remove("imc-ok", "imc-alerta", "imc-perigo");

    if (imc < 18.5) {
        resultadoDiv.innerHTML = "Seu IMC é: " + imc.toFixed(2) + " - Abaixo do peso.";
        resultadoDiv.classList.add("Abaixo do peso");
    } else if (imc >= 18.5 && imc < 24.9) {
        resultadoDiv.innerHTML = "Seu IMC é: " + imc.toFixed(2) + " - Peso normal.";
        resultadoDiv.classList.add("Peso normal");
    } else if (imc >= 25 && imc < 29.9) {
        resultadoDiv.innerHTML = "Seu IMC é: " + imc.toFixed(2) + " - Sobrepeso.";
        resultadoDiv.classList.add("Sobrepeso");
    } else {
        resultadoDiv.innerHTML = "Seu IMC é: " + imc.toFixed(2) + " - Obesidade.";
        resultadoDiv.classList.add("Obesidade");
    }
}
